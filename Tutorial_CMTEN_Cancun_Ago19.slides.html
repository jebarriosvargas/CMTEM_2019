1. Introduction to tight binding hamiltonian(TBH)¶
LEF Foa et al. D. Vanderbilt

Electronic Hamiltonian¶

$ \begin{align} H = T_{\rm e} + T_{\rm n} + V_{\rm e-e} + V_{\rm n-e} + V_{\rm n-n} \end{align} $

Adiabatic aproximation,

$ \begin{align} H_{\rm elec} = T_{\rm e} + V_{\rm n-e} + V_{\rm e-e} \end{align} $

Typically, we define

$ \begin{align} V_{\rm ext} = \sum_{\rm I} \frac{e^2 Z_I}{|\vec{r}-\vec{R}_I|} \end{align} $

then, we can write the electronic Hamiltonian as

$ \begin{align} H_{\rm elec} = \sum_{i} \bigg[ \frac{p_i^2}{2m} + V_{\rm ext}(\vec{r}_i) \bigg] + \frac{1}{2}\sum_{ij} \frac{e^2}{|\vec{r}_i-\vec{r}_j|} \end{align} $

Last equation is famously impossible to solve when the number of electros $N$ becomes large.

Crystal Hamiltonian¶

$ \begin{align} H = \frac{p^2}{2m} + V(\vec{r}) \end{align} $ where $V(\vec{r}-\vec{R}_\ell) = V(\vec{r})$ for any lattice vector $\vec{R}_\ell=\ell_1\vec{a}_1+\ell_2\vec{a}_2+\ell_3\vec{a}_3$.

$ \begin{align} H_{\vec{k}}=e^{-i\vec{k}\cdot\vec{r}} H e^{i\vec{k}\cdot\vec{r}} \end{align} $

$ \begin{align} \vec{v}=\frac{1}{i\hbar}[\vec{r},H] \longrightarrow \vec{v}=\frac{1}{\hbar}\nabla_{\vec{k}}H_{\vec{k}} \end{align} $

Tight-binding Hamiltonian¶

Core: Simplified representation that provies physical and chemical understanding.

Frequently, we choose some atomic-like orbitals $\varphi_j$ on site $j$. We can write the the trial Hamiltonian eigenstate as,

$ \begin{align} \psi_n(\vec{r}) = \sum_j C_{nj}\varphi_j(\vec{r}-\vec{r}_j) \end{align} $

where the $C_{nj}$, the expansion coefficients of the $n$th eigenstate on the $j$th basis orbital, are to be determined. Therefore, the solutions are given by solving the matrix equation

$ \begin{align} \big( H - E_n S) C_n = 0 \end{align} $

where $H$ and $S$ are the $M\times M$ matrices

$ \begin{align} H_{ij} &= \langle \varphi_i| H |\varphi_j\rangle \\ S_{ij} &= \langle \varphi_i| \varphi_j\rangle \end{align} $

$C_n$ is the $M$-component column vector of coefficients $C_{nj}$.

$H_{ii}$: site energies

$H_{ij}$: hopping matrix elements (off-diagonal)

$S_{ij}$: overlap matrix

2. Green's Functions in Quantum Physics¶
E.N. Economou D. Vanderbilt

$ \begin{eqnarray} \big[z-\mathcal{H}\big] G(z) = 1 \end{eqnarray} $

$ \begin{eqnarray} \big[E+i\eta-\mathcal{H}\big] G(E+i\eta) = 1\\ G^+(E) = \lim_{\eta\to 0^+} G(E+i\eta)\\ G^-(E) = \lim_{\eta\to 0^-} G(E+i\eta)\\ \end{eqnarray} $

$ \begin{align} {\rm DOS(E)} = \mp \frac{1}{\pi} {\rm Im} \big\{ {\rm Tr}\, G^{\pm}(E) \big\} \end{align} $

Delta function in terms of Chebyshev polynomials¶

$ \begin{align} f(x)\approx \frac{1}{\pi\sqrt{1-x^2}}\bigg(\mu_0+2\sum_{n=1}^{N-1}\mu_n T_n(x) \bigg)\,,\quad \mu_n=\int_{-1}^1 f(x) T_n(x)\,dx \end{align} $

Let $f(x)=\delta(x)$, then $\mu_n=T_n(0)$

Density of states¶

$ \begin{align} \delta(E)\approx \frac{1}{\pi\sqrt{1-E^2}}\bigg(\mu_0+2\sum_{n=1}^{N-1}\mu_n T_n(E) \bigg) \end{align} $

$ \begin{align} \mu_n = {\rm Tr} \bigg\{ T_n\big[ \mathcal{H}(\vec{k}) \big]\bigg\} \end{align} $

where $ \begin{align} \mathcal{H}(\vec{k}) = \left( \begin{array}{ccc} \ddots&&\\ &E(\vec{k})& \\ &&\ddots \end{array} \right)\,. \end{align} $

Recursive relation of Chebyshev polynomials}¶

Goal: Evaluate $\mu_n$,

$\begin{align} \mu_n = {\rm Tr} \bigg\{ T_n\big[ \mathcal{H}(\vec{k}) \big]\bigg\} \end{align}$

We wil use the recursive relation,

$ \begin{align} T_0(\mathcal{H}) &= \mathcal{I}\\ T_1(\mathcal{H}) &= \mathcal{H}\\ T_n(\mathcal{H}) &= 2\mathcal{H}T_{m-1}(\mathcal{H})-T_{m-2}(\mathcal{H}) \end{align} $

and the stochastic evaluation of traces,

$ \begin{align} {\rm Tr} \approx \frac{1}{R} \sum_{r=0}^{R-1} \langle r | T_n(\mathcal{H})|r\rangle \end{align} $

$ \begin{align} {\rm Tr}\bigg\{ T_n\big[ \mathcal{H}(\vec{k}) \big]\bigg\} \approx \frac{1}{R} \sum_{r=0}^{R-1} \langle r | T_n(\mathcal{H})|r\rangle \end{align} $

Algorithm: recursive relation of Chebyshev polynomials.

$ \begin{align} |r_0\rangle &= T_0(\mathcal{H}) |\Psi_{\rm rand}\rangle = |\Psi_{\rm rand}\rangle &\\ |r_1\rangle &= T_1(\mathcal{H}) |r_0 \rangle = \mathcal{H}|\Psi_{\rm rand}\rangle &\\ |r_2\rangle &= T_2(\mathcal{H}) |r_0\rangle = 2\mathcal{H}T_1|r_0\rangle-T_0(\mathcal{H})|r_0\rangle=2H|r_1\rangle+|r_0\rangle & \\ |r_3\rangle &= T_3(\mathcal{H}) |r_0\rangle = 2\mathcal{H}T_2|r_0\rangle-T_1(\mathcal{H})|r_0\rangle=2H|r_2\rangle+|r_1\rangle & \\ \vdots \\ |r_n\rangle &= T_n(\mathcal{H}) |r_0\rangle = 2\mathcal{H}T_{n-1}|r_0\rangle-T_{n-2}(\mathcal{H})|r_0\rangle=2H|r_{n-1}\rangle+|r_{n-2}\rangle & \end{align} $

$ \begin{align} |\Psi_{\rm rand}\rangle = \sum_i e^{i\theta_{\rm rand}} |i\rangle \end{align} $

Thumb rule¶

$ \begin{align} \bigg|\,f(E)-f_{\rm KPM}\,\bigg| \propto \frac{1}{\sqrt{N}} \end{align} $

Kubo-Greenwood formula¶

Linear response (electrical conductivity)¶

$ \begin{align} \langle j_x(0),j_x(t) \rangle \longrightarrow \sigma_{xx}=\frac{2\pi e^2\hbar^2}{\Omega} {\rm Tr} \bigg\{ \hat{v}_x \delta(E-\mathcal{H}) \hat{v}_x \delta(E-\mathcal{H}) \bigg\} \end{align} $

where $\hat{v}_x=[\mathcal{X},\mathcal{H}]/i\hbar$

$ \begin{align} \delta(E-\mathcal{H}) \longrightarrow {\rm expansion} \quad T_n(\mathcal{H}) \end{align} $

$ \begin{align} \sigma(E) = \frac{4e^2\hbar }{\pi\Omega} \sum_{n=0}^{\mathcal{N}-1} \sum_{m=0}^{\mathcal{M}-1} \mathcal{T}_{mn}(E)\, \mu_{mn}\,, \end{align} $

where

$ \begin{align} \mu_{mn} &= {\rm Tr}\bigg\{ \hat{v}_x T_n(\hat{h}) \hat{v}_x T_m(\hat{h}) \bigg\} = {\rm Tr}\bigg\{ \frac{a_{\rm cutoff}}{ \hbar} [\hat{h},\mathcal{X}]\, T_n(\hat{h}) \frac{a_{\rm cutoff}}{\hbar} [\hat{h},\mathcal{X}]\, T_m(\hat{h}) \bigg\} \end{align} $

using $\hat{h} = \mathcal{H}/W$ ($W$ half-bandwidth) and $\mathcal{X}=X/a_{\rm cutoff}$. $T_n$ are the Chebyshev polynomials. Therefore,

$ \begin{align} \sigma(E) = \frac{4e^2}{\pi h} \bigg( \frac{4\pi}{3\sqrt{3}}\frac{a_{\rm cutoff}^2}{a_{\rm CC}^2} \bigg) \sum_{n=0}^{\mathcal{N}-1} \sum_{m=0}^{\mathcal{M}-1} \mathcal{T}_{mn}(E)\, {\rm Tr}\bigg\{ [\hat{h},\mathcal{X}]\, T_n(\hat{h}) [\hat{h},\mathcal{X}]\, T_m(\hat{h}) \bigg\}\,. \end{align} $

The different energy expansion are,

$ \begin{align} &{\rm CPGF:}\quad \mathcal{T}_{mn}(z) = {\rm Im}\bigg[ \frac{-2i}{1+\delta n0}\frac{(z-i\sqrt{1-z^2})^n}{\sqrt{1-z^2}}\bigg] {\rm Im}\bigg[ \frac{-2i}{1+\delta m0}\frac{(z-i\sqrt{1-z^2})^m}{\sqrt{1-z^2}} \bigg]&\,,\\ &{\rm KPM:}\quad\,\, \mathcal{T}_{mn}(E) = \frac{g_n T_n(E) g_m T_m(E)}{1-E^2}& \end{align} $

Jackson Kernel:

$ \begin{align} g_n=\frac{1}{\mathcal{N}+1}\bigg[ (\mathcal{N}-n+1)\cos\frac{\pi n}{\mathcal{N}+1} +\sin\frac{\pi n}{\mathcal{N}+1}\cos\frac{\pi}{\mathcal{N}+1} \bigg]\,, \end{align} $

Lorentz Kernel:

$ \begin{align} g_n = \frac{ \sinh(\lambda (1-n/\mathcal{N}) }{\sinh(\lambda)}\,. \end{align} $

Single energy algorithm ($\sigma_{xx}$)¶

$ \begin{align} \mu_{mn} &= {\rm Tr}\bigg\{ \hat{v}_x T_n(\hat{h}) \quad \hat{v}_x T_m(\hat{h}) \bigg\} \end{align} $

$ \begin{align} |r\rangle_n &= T_n(\hat{h}) |\Psi_{\rm rand}\rangle \\ |\bar{r}\rangle_n &= T_n(\hat{h}) \hat{v}_x|\Psi_{\rm rand}\rangle \end{align} $

$ \begin{align} \sigma_{xx}(E) = \frac{4e^2}{\pi h} \bigg( \frac{4\pi}{3\sqrt{3}}\frac{a_{\rm cutoff}^2}{a_{\rm CC}^2} \bigg) \frac{1}{R}\sum_{r=1}^R\langle \varphi_{-}^{(r)}(E) | \varphi_{+}^{(r)}(E) \rangle \end{align} $

$\begin{align} | \varphi_{+}^{(r)}(E)\rangle = \sum_{n=0}^{\mathcal{N}-1} g_n(E)T_n(E) \hat{v}_x |r\rangle_n \end{align}$

$\begin{align} | \varphi_{-}^{(r)}(E)\rangle = \sum_{n=0}^{\mathcal{N}-1} g_n(E)T_n(E) |\bar{r}\rangle_n \end{align}$
